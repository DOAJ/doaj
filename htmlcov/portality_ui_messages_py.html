@classmethod 

def flash(cls, tup): 

if isinstance(tup, tuple): 

flash(tup[0], tup[1]) 

else: 

flash(tup) 

 

@classmethod 

def flash_with_url(cls, message, category): 

flash(message, category + '+contains-url')
